 01.Find average, maximum, minimum salary of the employees.

 SELECT empno, ROUND(MAX(sal),0) "Maximum",
ROUND(MIN(sal),0) "Minimum",
ROUND(AVG(sal),0) "Average"
FROM emp
GROUP BY empno;

 02.Find average, maximum, minimum salary of the employees according to department number.
  
  SELECT deptno, ROUND(MAX(sal),0) "Maximum",
ROUND(MIN(sal),0) "Minimum",
ROUND(AVG(sal),0) "Average"
FROM emp
GROUP BY deptno;

 03.Find average, maximum, minimum salary of the employees according to job category.

  SELECT job, ROUND(MAX(sal),0) "Maximum",
ROUND(MIN(sal),0) "Minimum",
ROUND(AVG(sal),0) "Average"
FROM emp
GROUP BY job;
