int top=-1;

void push(int item){

if(top==MAXSIZE-1)
{

   cout<<"Stack Overflow"<<endl;

return;
}
else
{
	top++;
	A[top]=item;
	cout<<"succesfully inserted"<<endl;
	
}
}

void pop()
{
	int item;
	if(top==-1)
	{
		cout<<"stack underflow"<<endl;
		return;
	}
else
{
	
	
	item=A[top];
	top--;
	cout<<"The deleted element is:"<<A[top--]<<endl;
}	
	
}
void show()
{ 
	int i;
	for(i=top;i>=0;i--)
